Что ты получаешь в итоге? Потерю компьютера или серьёзные затраты на ремонт, который обойдётся во много раз дороже, чем сама игра.

Время — тоже ресурс. Зачем тратить часы и нервы на борьбу с последствиями, если можно изначально избежать проблемы, купив лицензию?

  <script>
    const commentsDiv = document.getElementById('comments');

    // Загружаем сохранённые комментарии
    window.onload = () => {
      const saved = JSON.parse(localStorage.getItem('comments')) || [];
      saved.forEach(c => showComment(c));
    };

    function addComment() {
      const username = document.getElementById('username').value.trim();
      const text = document.getElementById('commentInput').value.trim();
      if (!username || !text) return alert('Введите ник и комментарий!');
      const comment = { id: Date.now(), username, text, likes: 0, dislikes: 0 };
      showComment(comment);
      saveComment(comment);
      document.getElementById('commentInput').value = '';
    }

    function showComment(comment) {
      const div = document.createElement('div');
      div.className = 'comment';
      div.innerHTML = `<b>${comment.username}</b>: ${comment.text}
        <div class="vote">
          <button onclick="vote(${comment.id}, 'like', this)">👍 ${comment.likes}</button>
          <button onclick="vote(${comment.id}, 'dislike', this)">👎 ${comment.dislikes}</button>
        </div>`;
      commentsDiv.appendChild(div);
    }

    function saveComment(comment) {
      const saved = JSON.parse(localStorage.getItem('comments')) || [];
      saved.push(comment);
      localStorage.setItem('comments', JSON.stringify(saved));
    }

    function vote(id, type, btn) {
      let voted = JSON.parse(localStorage.getItem('voted')) || {};
      if (voted[id]) {
        alert('Вы уже проголосовали за этот комментарий!');
        return;
      }
      let comments = JSON.parse(localStorage.getItem('comments')) || [];
      comments = comments.map(c => {
        if (c.id === id) {
          if (type === 'like') c.likes++;
          else c.dislikes++;
        }
        return c;
      });
      localStorage.setItem('comments', JSON.stringify(comments));
      voted[id] = type;
      localStorage.setItem('voted', JSON.stringify(voted));
      location.reload();
    }
  </script>
